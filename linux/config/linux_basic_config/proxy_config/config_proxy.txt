/etc/apt/apt.conf


Acquire::http::Proxy "http://192.168.x.x:8080/";
 
Acquire::https::Proxy "http://192.168.x.x:8080/";
 
Acquire::ftp::Proxy "http://192.168.x.x:8080/";
